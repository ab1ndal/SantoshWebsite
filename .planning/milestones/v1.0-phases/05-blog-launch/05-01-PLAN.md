       ```typescript
       import { structureTool } from 'sanity/structure'
       import { postSchema } from './src/lib/sanity/schemas/post'

       export default defineConfig({
         name: 'santosh-studio',

         projectId: process.env.NEXT_PUBLIC_SANITY_PROJECT_ID!,

           types: [postSchema],
         },
       })
       ```
       Note per D-02: Use `NEXT_PUBLIC_SANITY_PROJECT_ID` (not server-only) because sanity.config.ts is imported by the Studio component which runs client-side (Pitfall 4 from RESEARCH.md). Do NOT include `visionTool` — it requires an additional dependency not in the install list.

    2. Create `src/app/studio/[[...tool]]/page.tsx`:
       ```typescript
       'use client'
       import { NextStudio } from 'next-sanity/studio'
       import config from '../../../../sanity.config'

       export default function StudioPage() {
         return <NextStudio config={config} />
       }
       ```
       Note: Per Pitfall 3 from RESEARCH.md, the `dynamic` and `viewport` exports must be separated. Create a companion `src/app/studio/[[...tool]]/layout.tsx` file that exports `dynamic = 'force-dynamic'` OR add `export const dynamic = 'force-dynamic'` to the page file. Since NextStudio is a client component, the `dynamic` export can go in a separate server layout or the page itself.

       Actually, per the research code example: put `export const dynamic = 'force-dynamic'` and `export { viewport } from 'next-sanity/studio'` in the page file alongside the default export. However, since the page has `'use client'`, these server-side exports may not work. Create a separate `src/app/studio/[[...tool]]/layout.tsx` (Server Component, no 'use client') that exports `dynamic` and `viewport`:

       `src/app/studio/[[...tool]]/layout.tsx`:
       ```typescript
       export const dynamic = 'force-dynamic'
       export { viewport } from 'next-sanity/studio'

       export default function StudioLayout({ children }: { children: React.ReactNode }) {
         return children
       }
       ```
       This ensures Studio is never statically pre-rendered (Pitfall 3) while keeping the page as a client component.
